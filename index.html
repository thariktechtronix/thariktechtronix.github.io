Desk Buddy Robot - Episode 2

Welcome to the official project page! In this build, we upgraded the brain to an ESP32, added a joystick, built a PIR sensor sleep mode, and hid a fully playable Snake Game inside the OLED display.

Circuit & Pin Connections

Follow this exact wiring guide to connect the components to your ESP32 Development Board.

OLED Display (I2C)SDA / SCLGPIO 21 / GPIO 22
MPU6050 GyroscopeSDA / SCLGPIO 21 / GPIO 22
Analog JoystickVRx / VRy / SWGPIO 32 / GPIO 33 / GPIO 15
PIR Motion SensorOUT (Data)GPIO 13
Passive BuzzerPositive (+)GPIO 4

Master Source Code

Upload this directly to your ESP32. Make sure you have installed the Adafruit GFX, Adafruit SSD1306, and MPU6050 libraries.

DeskBuddy_Ep2_Master.ino 
#include <Wire.h>
#include <Adafruit_GFX.h>
#include <Adafruit_SSD1306.h>
#include <MPU6050.h>
#include <WiFi.h>          
#include <HTTPClient.h>    
#include <ArduinoJson.h>
#include <time.h>

// --- USER CONFIGURATION ---
const char* ssid = "AKMADTHARIK";         
const char* password = "akmad1234"; 
const char* weatherKey = "b820559e7f1b40ee1d047e7ffa2d4a65"; 
const char* city = "Salem,IN"; 

// --- ESP32 PINS ---
#define BUTTON_PIN 15  
#define BUZZER_PIN 4
#define PIR_PIN 13
#define JOY_X 32       
#define JOY_Y 33       

#define SCREEN_WIDTH 128
#define SCREEN_HEIGHT 64

Adafruit_SSD1306 display(SCREEN_WIDTH, SCREEN_HEIGHT, &Wire, -1);
MPU6050 mpu;

// --- STATE MACHINE ---
enum ScreenState { FACE_DEFAULT, FACE_TRACKING, FACE_HAPPY, FACE_ERROR, PAGE_TIME, PAGE_WEATHER_TODAY, PAGE_WEATHER_TOMORROW, PAGE_GAME };
ScreenState currentState = FACE_TRACKING; 

// --- VARIABLES ---
int16_t ax, ay, az;
float sensitivity = 0.0035;
float smoothX = 0, smoothY = 0;

bool joyMoved = false;
bool btnHandled = false;

String currentTemp = "--";
unsigned long lastWeatherUpdate = 0;

unsigned long lastMotionTime = 0;
const unsigned long sleepDelay = 30000; // 30 SECONDS SLEEP TIMER
bool isAsleep = false;
unsigned long lastBlink = 0;
bool isBlinking = false;

// --- SNAKE GAME VARIABLES ---
int snakeX[20], snakeY[20];
int snakeLen = 3;
int snakeDir = 1;       // 0:Up, 1:Right, 2:Down, 3:Left
int currentMoveDir = 1; 
int foodX, foodY;
bool gameOver = false;
bool gameStarted = false; 
unsigned long lastSnakeMove = 0;

// --- FORWARD DECLARATIONS ---
void playTransition(); void fetchWeather(); void drawBootScreen(); void drawCustomIntro(); 
void drawDefaultFace(); void drawTrackingFace(); void drawHappyFace(); void drawErrorFace(); 
void drawTimeDate(); void drawWeatherToday(); void drawWeatherTomorrow(); void drawSnakeGame(); void resetGame();

// --- SOUND EFFECTS ---
void beepClick() { tone(BUZZER_PIN, 1500, 50); }
void beepStartup() { tone(BUZZER_PIN, 800, 100); delay(100); tone(BUZZER_PIN, 1200, 100); delay(100); tone(BUZZER_PIN, 1800, 200); }
void beepHappy() { tone(BUZZER_PIN, 2000, 100); delay(100); tone(BUZZER_PIN, 2500, 150); }
void beepSad() { tone(BUZZER_PIN, 400, 200); delay(200); tone(BUZZER_PIN, 300, 400); }
void beepEat() { tone(BUZZER_PIN, 3000, 30); }

// --- SETUP ---
void setup() {
  Serial.begin(115200);
  Wire.begin(21, 22); 
  
  pinMode(BUTTON_PIN, INPUT_PULLUP);
  pinMode(BUZZER_PIN, OUTPUT);
  pinMode(PIR_PIN, INPUT);
  analogReadResolution(10); // 10-bit joystick fix

  display.begin(SSD1306_SWITCHCAPVCC, 0x3C);
  display.setTextColor(WHITE);
  
  drawBootScreen();
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) { delay(500); }
  configTime(19800, 0, "pool.ntp.org", "time.nist.gov"); 
  fetchWeather(); 

  mpu.initialize();
  resetGame(); 
  drawCustomIntro();
  lastMotionTime = millis(); 
}

// --- MAIN LOOP ---
void loop() {
  checkMotion();

  if (!isAsleep) {
    handleButton(); 
    switch (currentState) {
      case FACE_DEFAULT:          drawDefaultFace(); break;
      case FACE_TRACKING:         drawTrackingFace(); break;
      case FACE_HAPPY:            drawHappyFace(); break;
      case FACE_ERROR:            drawErrorFace(); break;
      case PAGE_TIME:             drawTimeDate(); break;
      case PAGE_WEATHER_TODAY:    drawWeatherToday(); break;
      case PAGE_WEATHER_TOMORROW: drawWeatherTomorrow(); break;
      case PAGE_GAME:             drawSnakeGame(); break; 
    }
  }
  delay(10); 
}

// --- SMOOTH TRANSITION ANIMATION ---
void playTransition() {
  for (int i = 0; i < 32; i += 4) {
    display.clearDisplay();
    display.fillRect(0, i, 128, 64 - (i * 2), WHITE);
    display.display();
    delay(15); 
  }
  display.clearDisplay(); display.display();
}

// --- 30 SECOND SLEEP & WAKE LOGIC ---
void checkMotion() {
  int pirState = digitalRead(PIR_PIN);
  if (pirState == HIGH) {
    lastMotionTime = millis(); 
    if (isAsleep) {
      isAsleep = false;
      display.ssd1306_command(SSD1306_DISPLAYON); 
      delay(100); 
      beepHappy();
      drawHappyFace(); // Wakes up with a big smile!
      delay(2000); 
    }
  }

  if (!isAsleep && (millis() - lastMotionTime > sleepDelay)) {
    isAsleep = true;
    beepSad();
    display.clearDisplay(); display.display();
    display.ssd1306_command(SSD1306_DISPLAYOFF); 
  }
}

// --- JOYSTICK LOGIC ---
void handleButton() {
  int xVal = analogRead(JOY_X); 
  int yVal = analogRead(JOY_Y); 
  bool btnPressed = (digitalRead(BUTTON_PIN) == LOW); 

  if (currentState == PAGE_GAME) {
    if (btnPressed && !btnHandled) {
      btnHandled = true;
      if (gameOver) { resetGame(); beepClick(); } 
      else { currentState = PAGE_TIME; playTransition(); }
    } else if (!btnPressed) { btnHandled = false; }
    return; 
  }

  // Smooth Menu Scrolling (Y-Axis)
  if (yVal < 200 && !joyMoved) {
    beepClick(); playTransition(); currentState = static_cast<ScreenState>((currentState + 1) % 8); joyMoved = true; 
    if (currentState == FACE_ERROR) beepSad(); if (currentState == FACE_HAPPY) beepHappy();
  } 
  else if (yVal > 800 && !joyMoved) {
    beepClick(); playTransition(); currentState = static_cast<ScreenState>((currentState + 7) % 8); joyMoved = true;
    if (currentState == FACE_ERROR) beepSad(); if (currentState == FACE_HAPPY) beepHappy();
  } 
  else if (yVal >= 200 && yVal <= 800 && xVal >= 200 && xVal <= 800) { joyMoved = false; }

  // Petting (Click Joystick for Big Happy Face)
  if (btnPressed && !btnHandled) {
    currentState = FACE_HAPPY; beepHappy(); btnHandled = true;
  } else if (!btnPressed) { btnHandled = false; }
}

// --- BULLETPROOF SNAKE GAME ---
void resetGame() {
  snakeLen = 3; snakeDir = 1; currentMoveDir = 1; 
  snakeX[0] = 16; snakeY[0] = 8; snakeX[1] = 15; snakeY[1] = 8; snakeX[2] = 14; snakeY[2] = 8;
  foodX = random(2, 29); foodY = random(2, 13);
  gameOver = false; gameStarted = false; lastSnakeMove = millis(); 
}

void drawSnakeGame() {
  display.clearDisplay();

  if (gameOver) {
    display.setTextSize(2); display.setCursor(10, 15); display.print("GAME OVER");
    display.setTextSize(1); display.setCursor(15, 45); display.print("Click to Restart");
    display.display();
    return;
  }

  int xVal = analogRead(JOY_X); int yVal = analogRead(JOY_Y);
  int dx = abs(xVal - 511); int dy = abs(yVal - 511);

  if (dx > 350 || dy > 350) {
    gameStarted = true;
    if (dx > dy) { 
      if (xVal < 150 && currentMoveDir != 2) snakeDir = 0;       // Up
      else if (xVal > 850 && currentMoveDir != 0) snakeDir = 2;  // Down 
    } else { 
      if (yVal > 850 && currentMoveDir != 1) snakeDir = 3;       // Left
      else if (yVal < 150 && currentMoveDir != 3) snakeDir = 1;  // Right
    }
  }

  if (!gameStarted) { display.setTextSize(1); display.setCursor(25, 30); display.print("Move to Start"); }

  if (gameStarted && (millis() - lastSnakeMove > 150)) {
    lastSnakeMove = millis();
    currentMoveDir = snakeDir; 

    for (int i = snakeLen - 1; i > 0; i--) { snakeX[i] = snakeX[i - 1]; snakeY[i] = snakeY[i - 1]; }
    
    if (snakeDir == 0) snakeY[0]--; else if (snakeDir == 1) snakeX[0]++;
    else if (snakeDir == 2) snakeY[0]++; else if (snakeDir == 3) snakeX[0]--;

    if (snakeX[0] < 0 || snakeX[0] >= 32 || snakeY[0] < 0 || snakeY[0] >= 16) { gameOver = true; beepSad(); }
    for (int i = 1; i < snakeLen; i++) { if (snakeX[0] == snakeX[i] && snakeY[0] == snakeY[i]) { gameOver = true; beepSad(); } }

    if (snakeX[0] == foodX && snakeY[0] == foodY) {
      if (snakeLen < 20) snakeLen++; foodX = random(2, 29); foodY = random(2, 13); beepEat();
    }
  }

  display.fillRect(foodX * 4, foodY * 4, 4, 4, WHITE);
  for (int i = 0; i < snakeLen; i++) { display.fillRect(snakeX[i] * 4, snakeY[i] * 4, 4, 4, WHITE); }
  display.drawRect(0, 0, 128, 64, WHITE); display.display();
}

// --- CUTER EMO FACES & SCREENS ---
void drawBootScreen() { display.clearDisplay(); display.setTextSize(1); display.setCursor(25, 20); display.print("CONNECTING"); display.setCursor(25, 35); display.setTextSize(2); display.print("WiFi..."); display.display(); }
void drawCustomIntro() { display.clearDisplay(); display.setTextSize(2); display.setCursor(28, 15); display.print("THARIK"); display.setTextSize(1); display.setCursor(34, 38); display.print("TECHTRONIX"); display.display(); beepStartup(); delay(2000); playTransition(); }

void drawDefaultFace() {
  display.clearDisplay();
  if (millis() - lastBlink > random(3000, 6000)) { isBlinking = true; lastBlink = millis(); }
  if (isBlinking && millis() - lastBlink > 150) { isBlinking = false; }
  
  if (isBlinking) { display.fillRoundRect(22, 28, 36, 6, 4, WHITE); display.fillRoundRect(70, 28, 36, 6, 4, WHITE); } 
  else {
    display.fillRoundRect(20, 12, 40, 40, 12, WHITE); display.fillRoundRect(68, 12, 40, 40, 12, WHITE);
    display.fillRoundRect(34, 24, 12, 12, 4, BLACK); display.fillRoundRect(82, 24, 12, 12, 4, BLACK);
  }
  display.display();
}

void drawTrackingFace() {
  mpu.getAcceleration(&ax, &ay, &az);
  smoothX = smoothX * 0.3 + (ay * sensitivity) * 0.7; smoothY = smoothY * 0.3 + (ax * sensitivity) * 0.7;
  display.clearDisplay();
  display.fillRoundRect(20, 12, 40, 40, 12, WHITE); display.fillRoundRect(68, 12, 40, 40, 12, WHITE);
  int ox = constrain((int)smoothX, -10, 10); int oy = constrain((int)smoothY, -10, 10);
  display.fillRoundRect(34 + ox, 24 + oy, 12, 12, 4, BLACK); display.fillRoundRect(82 + ox, 24 + oy, 12, 12, 4, BLACK);
  display.display();
}

void drawHappyFace() {
  display.clearDisplay();
  display.fillRoundRect(20, 12, 40, 40, 12, WHITE); display.fillRoundRect(68, 12, 40, 40, 12, WHITE);
  display.fillRect(18, 28, 92, 24, BLACK); 
  display.fillCircle(64, 44, 14, WHITE); display.fillCircle(64, 38, 16, BLACK); 
  display.display();
}

void drawErrorFace() { display.clearDisplay(); display.fillRoundRect(22, 16, 36, 32, 8, WHITE); display.fillRoundRect(70, 16, 36, 32, 8, WHITE); display.fillTriangle(15, 10, 65, 10, 65, 30, BLACK); display.fillTriangle(63, 10, 113, 10, 63, 30, BLACK); display.fillRoundRect(38, 30, 12, 12, 3, BLACK); display.fillRoundRect(78, 30, 12, 12, 3, BLACK); display.fillRoundRect(56, 52, 16, 4, 2, WHITE); display.display(); }
void drawTimeDate() { display.clearDisplay(); time_t now = time(nullptr); struct tm* p = localtime(&now); char tBuf[10], dBuf[20]; strftime(tBuf, sizeof(tBuf), "%H:%M", p); strftime(dBuf, sizeof(dBuf), "%d %b %Y", p); display.setTextSize(3); display.setCursor(20, 10); display.print(tBuf); display.setTextSize(1); display.setCursor(30, 45); display.print(dBuf); display.display(); }
void drawWeatherToday() { if (millis() - lastWeatherUpdate > 600000) fetchWeather(); display.clearDisplay(); display.drawCircle(25, 32, 10, WHITE); display.drawLine(25, 15, 25, 20, WHITE); display.drawLine(25, 44, 25, 49, WHITE); display.drawLine(8, 32, 13, 32, WHITE); display.drawLine(37, 32, 42, 32, WHITE); display.drawLine(13, 20, 16, 23, WHITE); display.drawLine(34, 41, 37, 44, WHITE); display.drawLine(13, 44, 16, 41, WHITE); display.drawLine(34, 23, 37, 20, WHITE); display.setTextSize(2); display.setCursor(60, 20); display.print(currentTemp); display.setTextSize(1); display.setCursor(60, 45); display.print("SALEM | NOW"); display.display(); }
void drawWeatherTomorrow() { display.clearDisplay(); display.drawCircle(20, 25, 6, WHITE); display.drawLine(20, 15, 20, 17, WHITE); display.fillCircle(25, 40, 8, WHITE); display.fillCircle(35, 35, 10, WHITE); display.fillCircle(45, 40, 8, WHITE); display.fillRect(25, 40, 20, 8, WHITE); display.setTextSize(1); display.setCursor(60, 20); display.print("28C - 33C"); display.setCursor(60, 45); display.print("SALEM | FCST"); display.display(); }

void fetchWeather() { if (WiFi.status() == WL_CONNECTED) { HTTPClient http; http.begin("http://api.openweathermap.org/data/2.5/weather?q=" + String(city) + "&units=metric&appid=" + String(weatherKey)); if (http.GET() > 0) { DynamicJsonDocument doc(1024); deserializeJson(doc, http.getString()); currentTemp = String((int)doc["main"]["temp"]) + "C"; lastWeatherUpdate = millis(); } http.end(); } }
